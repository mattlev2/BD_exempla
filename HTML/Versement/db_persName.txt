CREATE TABLE exempla
(
  id              INT unsigned NOT NULL AUTO_INCREMENT, # ID unique
  identsource	   INT NOT NULL,					 # Identifiant de l’exemple (donné par le XML)
  name            VARCHAR(150),            		      # nom de l’exemple
  pers            VARCHAR(150),              	      # personnage / animal qui intervient
  nomenclature    TEXT,                    		      # nomenclature officielle du type d’exemple si elle existe
  PRIMARY KEY     (identsource)                         # Make the identsource the primary key
);
